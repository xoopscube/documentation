85 {
86 $dirlist = [];
87 if (is_dir($dirname) && $handle = opendir($dirname)) {
88 while (false !== ($file = readdir($handle))) {
89 if (!preg_match("/^\..*$/", $file)) {
90 if ('cvs' != strtolower($file) && is_dir($dirname . $file)) {
91 $dirlist[$file]=$file;
92 }
93 }
94 }
95 closedir($handle);
96 asort($dirlist);
97 reset($dirlist);
98 }
